FROM node:20-alpine

WORKDIR /app

# Copia primero los archivos de dependencias
COPY package.json package-lock.json* ./

# Si existe lock → npm ci    Si no existe → npm install
RUN if [ -f package-lock.json ]; then npm ci --prefer-offline --no-audit --fund=false; else npm install --prefer-offline --no-audit --fund=false; fi

# Copia el resto del proyecto
COPY . .

# Build (cambia "build" si tu script se llama distinto)
RUN npm run build

# Servir con nginx (asumiendo salida en /app/dist)
FROM nginx:alpine

COPY --from=0 /app/dist /usr/share/nginx/html

# Configuración mínima para SPA (react-router, vite, etc.)
COPY <<EOF /etc/nginx/conf.d/default.conf
server {
    listen 80;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    location / {
        try_files \$uri \$uri/ /index.html;
    }

    gzip on;
    gzip_types text/plain text/css application/json application/javascript;
}
EOF

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
